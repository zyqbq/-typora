        const dug = $el => {
            const startPosition = {};
            const movePosition = {};
            const curPosition = {
                x: 0,
                y: 0
            };
            $el.addEventListener('pointerdown', startHandler, false);
            $el.addEventListener('touchstart', evt => {
                evt.preventDefault()
            }, false);
            function startHandler(evt) {
                // evt.preventDefault();
                startPosition.x = evt.pageX;
                startPosition.y = evt.pageY;
                document.addEventListener('pointermove', moveHandler, false);
                document.addEventListener('pointerup', endHandler, false);
                document.addEventListener('pointercancel', endHandler, false);
            };
            function moveHandler(evt) {
                movePosition.x = curPosition.x + evt.pageX - startPosition.x;
                movePosition.y = curPosition.y + evt.pageY - startPosition.y;
                $el.style.transform = `translate3d(${movePosition.x}px,${movePosition.y}px,0)`;
            };
            function endHandler() {
                curPosition.x = movePosition.x;
                curPosition.y = movePosition.y
                document.removeEventListener('pointermove', moveHandler, false);
                document.removeEventListener('pointerup', endHandler, false);
                document.removeEventListener('pointercancel', endHandler, false);
            };
        }
        dug(document.getElementById('box'))
    </script>
</body>

</html>
